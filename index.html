<!-- MULTI-STEP CONTROLLER JS -->

<script>

let currentStep = 0;


function showStep(i) {

    const steps = document.querySelectorAll('.step');

    steps.forEach((step, index) => {

        step.style.display = index === i ? 'block' : 'none';

    });


    const progress = ((i + 1) / steps.length) * 100;

    document.getElementById('progressBar').style.width = progress + "%";


    window.scrollTo({ top: 0, behavior: 'smooth' });

}


function nextStep() {

    const steps = document.querySelectorAll('.step');

    if (currentStep < steps.length - 1) {

        currentStep++;

        showStep(currentStep);

    }

}


function prevStep() {

    if (currentStep > 0) {

        currentStep--;

        showStep(currentStep);

    }

}


document.addEventListener("DOMContentLoaded", () => showStep(0));

</script>

function generateReport() {


    // Collect all inputs

    const data = {

        practiceName: document.getElementById('practiceName').value || "Your Practice",

        practiceType: document.getElementById('practiceType').value,

        locations: parseInt(document.getElementById('locations').value) || 1,

        monthlyProduction: parseInt(document.getElementById('monthlyProduction').value) || 100000,

        activePatients: parseInt(document.getElementById('activePatients').value) || 2000,

        newPatientsPerMonth: parseInt(document.getElementById('newPatientsPerMonth').value) || 30,

        pmsSoftware: document.getElementById('pmsSoftware').value,

        staffCount: parseInt(document.getElementById('staffCount').value) || 8,


        appointmentReminders: document.getElementById('appointmentReminders').value,

        afterHoursScheduling: document.getElementById('afterHoursScheduling').value,

        recallSystem: document.getElementById('recallSystem').value,

        patientEducation: document.getElementById('patientEducation').value,

        reviewGeneration: document.getElementById('reviewGeneration').value,


        leadFollowUp: document.getElementById('leadFollowUp').value,

        treatmentPlanFollowUp: document.getElementById('treatmentPlanFollowUp').value,

        referralManagement: document.getElementById('referralManagement').value,

        noShowRate: document.getElementById('noShowRate').value,


        frontDeskTurnover: document.getElementById('frontDeskTurnover').value,

        trainingProcess: document.getElementById('trainingProcess').value,

        manualTaskLoad: document.getElementById('manualTaskLoad').value

    };


    // Recommendations engine

    const recs = { phase1: [], phase2: [], phase3: [] };


    // === Phase 1 – Foundation ===

    if (["manual","basic"].includes(data.appointmentReminders)) {

        recs.phase1.push({

            name: "Automated Appointment Reminders",

            description: "AI-driven SMS/email reminders with 2-way sync.",

            impact: "Reduces no-shows 25–35%"

        });

    }


    if (["no","voicemail"].includes(data.afterHoursScheduling)) {

        recs.phase1.push({

            name: "AI 24/7 Scheduling Assistant",

            description: "Enables after-hours booking without staff.",

            impact: "15–20% more appointments captured"

        });

    }


    if (["manual","inconsistent"].includes(data.recallSystem)) {

        recs.phase1.push({

            name: "Automated Recall System",

            description: "Hygiene and recare reminders sent automatically.",

            impact: "Boosts recall visits 30–40%"

        });

    }


    // === Phase 2 – Growth ===

    if (["poor","inconsistent"].includes(data.treatmentPlanFollowUp)) {

        recs.phase2.push({

            name: "Treatment Follow-Up Automation",

            description: "Automated sequences for unscheduled treatment.",

            impact: "Improves case acceptance 20–25%"

        });

    }


    if (["minimal","none","basic"].includes(data.patientEducation)) {

        recs.phase2.push({

            name: "Automated Patient Education",

            description: "AI sends procedure-specific explanations.",

            impact: "Improves acceptance & reduces patient anxiety"

        });

    }


    if (["none","manual"].includes(data.reviewGeneration)) {

        recs.phase2.push({

            name: "Review Generation System",

            description: "Automated review requests after visits.",

            impact: "Adds 40–60 reviews annually"

        });

    }


    if (["slow","inconsistent"].includes(data.leadFollowUp)) {

        recs.phase2.push({

            name: "AI Lead Follow-Up Automation",

            description: "Instant responses + multi-message nurturing.",

            impact: "Converts 30% more patient inquiries"

        });

    }


    // === Phase 3 – Optimization ===

    if (["poor","manual"].includes(data.referralManagement)) {

        recs.phase3.push({

            name: "Referral Partner Automation",

            description: "AI automates referral communication & updates.",

            impact: "Strengthens specialist relationships"

        });

    }


    // Always included

    recs.phase3.push({

        name: "Performance Analytics Dashboard",

        description: "Real-time reporting, KPIs & revenue tracking.",

        impact: "Gives data-driven clarity"

    });


    // Calculate totals & ROI

    const totalFeatures = recs.phase1.length + recs.phase2.length + recs.phase3.length;


    let monthlyFee, packageName;


    if (totalFeatures <= 4) {

        monthlyFee = 797 * data.locations;

        packageName = "Starter Package";

    } else if (totalFeatures <= 7) {

        monthlyFee = 1497 * data.locations;

        packageName = "Growth Package";

    } else {

        monthlyFee = 2297 * data.locations;

        packageName = "Complete System";

    }


    const setupFee = 1500 * data.locations;

    const totalFirstMonth = setupFee + monthlyFee;


    const estimatedIncrease = totalFeatures * 0.03; // 3% per improvement

    const projectedMonthlyIncrease = Math.round(data.monthlyProduction * estimatedIncrease);

    const projectedAnnualIncrease = projectedMonthlyIncrease * 12;

    const annualInvestment = monthlyFee * 12;


    const roi = Math.round(((projectedAnnualIncrease - annualInvestment) / annualInvestment) * 100);



    // =============================

    // BUILD REPORT HTML

    // =============================

    let html = `

        <div class="glass-card p-6 mt-8">


            <h2 class="text-3xl font-bold text-blue-400 mb-4">

                ${data.practiceName} — AI Implementation Plan

            </h2>


            <!-- Package Summary -->

            <div class="grid md:grid-cols-3 gap-6 mb-8">

                <div>

                    <p class="text-gray-400 text-sm">Recommended Package</p>

                    <p class="text-xl font-bold text-white">${packageName}</p>

                </div>

                <div>

                    <p class="text-gray-400 text-sm">Estimated Timeline</p>

                    <p class="text-xl font-bold text-white">

                        ${totalFeatures <= 4 ? "3–4 weeks" : totalFeatures <= 7 ? "6–8 weeks" : "10–12 weeks"}

                    </p>

                </div>

                <div>

                    <p class="text-gray-400 text-sm">Projected Annual ROI</p>

                    <p class="text-xl font-bold text-green-400">${roi}%</p>

                </div>

            </div>


            <!-- Revenue Projection -->

            <div class="glass-card p-6 border border-green-700 mb-8">

                <h3 class="text-xl font-semibold text-green-300 mb-4">Revenue Impact Projection</h3>

                <p class="text-green-400 text-3xl font-bold">$${projectedMonthlyIncrease.toLocaleString()} / month</p>

                <p class="text-green-400 text-xl font-bold mt-2">$${projectedAnnualIncrease.toLocaleString()} / year</p>

            </div>


            <!-- PHASES -->

    `;


    function phaseSection(title, color, items) {

        if (!items.length) return "";

        let section = `

            <div class="glass-card p-6 mb-8 border-l-4 border-${color}-400">

                <h3 class="text-xl font-bold text-${color}-300 mb-4">${title}</h3>

        `;

        items.forEach(item => {

            section += `

                <div class="mb-4 bg-gray-900 p-4 rounded border border-gray-700">

                    <p class="text-lg font-semibold text-white">${item.name}</p>

                    <p class="text-gray-300 text-sm">${item.description}</p>

                    <p class="text-${color}-300 text-sm mt-1">📈 ${item.impact}</p>

                </div>

            `;

        });

        return section + "</div>";

    }


    html += phaseSection("Phase 1 — Foundation", "blue", recs.phase1);

    html += phaseSection("Phase 2 — Growth", "purple", recs.phase2);

    html += phaseSection("Phase 3 — Optimization", "orange", recs.phase3);


    // Investment Summary

    html += `

        <div class="glass-card p-6 border border-gray-600">

            <h3 class="text-2xl font-bold text-white mb-4">Investment Summary</h3>

            <p class="text-gray-300 mb-2">Setup Fee: <span class="text-white font-bold">$${setupFee.toLocaleString()}</span></p>

            <p class="text-gray-300 mb-2">Monthly Fee: <span class="text-white font-bold">$${monthlyFee.toLocaleString()}</span></p>


            <p class="text-xl text-green-400 font-bold mt-4">

                Total First Month: $${totalFirstMonth.toLocaleString()}

            </p>

        </div>


        <div class="mt-6 flex gap-4 no-print">

            <button onclick="window.print()" 

                class="next-btn bg-green-600 hover:bg-green-700 flex-1">Print / Save as PDF</button>


            <button onclick="resetForm()" 

                class="back-btn flex-1">Start New Audit</button>

        </div>

    `;


    // Display report

    document.getElementById('formSection').style.display = "none";

    const results = document.getElementById('resultsSection');

    results.innerHTML = html;

    results.style.display = "block";


    window.scrollTo({ top: 0, behavior: "smooth" });

}



// Reset the audit

function resetForm() {

    document.getElementById('resultsSection').style.display = "none";

    document.getElementById('formSection').style.display = "block";

    currentStep = 0;

    showStep(0);

}

</script>

</body>

</html>
